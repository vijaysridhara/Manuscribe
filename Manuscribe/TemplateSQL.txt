CREATE TABLE "Content" (
	"id"	INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
	"parent" INTEGER NOT NULL DEFAULT -1,
	"seq"	INTEGER NOT NULL,
	"name"	VARCHAR(100),
	"level" CHAR(5) NOT NULL DEFAULT 'DRAFT',
	"status" CHAR(1) NOT NULL DEFAULT 'A',
	"content"	CLOB
);
CREATE TABLE "Details" (
	"title"	VARCHAR(100) NOT NULL,
	"subtitle"	VARCHAR(100) NOT NULL,
	"author"	VARCHAR(100) NOT NULL,
	"initdate"	DATE NOT NULL,
	"moddate"	DATE NOT NULL,
	PRIMARY KEY("title")
);
CREATE TABLE "Metadata" (
	"name"	VARCHAR(100) NOT NULL,
	"type"	CHAR(10) NOT NULL,
	"pagewidth" INTEGER NOT NULL,
	"pageheight" INTEGER NOT NULL,
	"left" INTEGER NOT NULL  DEFAULT 50,
	"top" INTEGER NOT NULL DEFAULT 50,
	"right" INTEGER NOT NULL DEFAULT 40,
	"bottom" INTEGER NOT NULL DEFAULT 40",
	"content"	CLOB
);
CREATE TABLE "Research" (
	"id"	INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
	"name"	VARCHAR(200),
	"parent" INTEGER NOT NULL DEFAULT -1,
	"type"	VARCHAR(10),
	"content"	BLOB
);
CREATE TABLE "Scenes" (
	"id"	INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
	"seq"	INTEGER NOT NULL,
	"name"	VARCHAR(100),
	"content"	CLOB
)
